const express = require('express'); const app = express(); const port = 3000; app.use(express.json()); app.get('/data', (req, res) => { const { lat, lng } = req.query; const username = 'kazzyyy'; const apiUrl = `http://api.geonames.org/findNearbyJSON?lat=${lat}&lng=${lng}&username=${username}`; // Using dynamic import() syntax import('node-fetch') .then(fetchModule => { const fetch = fetchModule.default; // Fetch data using the fetch function fetch(apiUrl) .then(response => response.json()) .then(data => { // Process the data as needed before sending it to the client res.json(data); }) .catch(error => { console.error('Error:', error); res.status(500).json({ error: 'Internal Server Error' }); }); }) .catch(error => { console.error('Error importing node-fetch:', error); res.status(500).json({ error: 'Internal Server Error' }); }); }); app.listen(port, () => { console.log(`Server is running on port ${port}`); });
